# This test tests the ComplexArrayMerge parser function

!! test

{{#complexarraydefine:left|a,b,c}}
{{#complexarraydefine:right|d,e,f}}
{{#complexarraymerge:example|left|right}}
{{#complexarrayprint:example}}
!! html
<p><br />
</p>
<ul><li>a</li>
<li>b</li>
<li>c</li>
<li>d</li>
<li>e</li>
<li>f</li></ul>

!! end
